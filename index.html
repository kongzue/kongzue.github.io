Kongzue.io

此处是一些关于开发的指导性意见。

大部分使用 Kongzue 目前已有的开源库，也有部分是比较好用的第三方库。

开始

打开 Android Studio，版本号3.1.2

框架

根据需求引入包如下：

dependencies {
    implementation fileTree(include: ['*.jar'], dir: 'libs')
//对话框
    implementation 'com.kongzue.dialog:dialog:2.3.4'
//基础框架
    implementation 'com.kongzue.baseframework:baseframework:6.5.7.1'
//是否需要更新组件及模块：
    implementation 'com.kongzue.kongzueupdatesdk:kongzueupdatesdk:1.4.0'
//是否需要选择图片、拍照的功能：
    implementation 'com.kongzue.takephoto:takephoto:2.0.4'
//若是需要上传功能，建议okHttp，否则推荐Volley
    implementation 'com.kongzue.baseokhttp:baseokhttp:2.1.2'
//若是需要底部导航栏
    implementation 'com.kongzue.tabbar:tabbar:1.1'
//若是需要顶部标题栏+沉浸式适配方案2
    implementation 'com.kongzue.titlebar:titlebar:1.0'
//若是需要数据库
    implementation 'com.kongzue.kongzuedb:kongzuedb:1.0.1'
//简单的图片加载：
    implementation 'com.facebook.fresco:fresco:0.12.0'
//是否需要视频播放：
    implementation 'com.shuyu:GSYVideoPlayer:4.1.2'
//根据需求是否需要侧滑返回：
    implementation 'me.majiajie:swipe-back:1.0.0-alpha2'
//是否需要轮播图：
    implementation 'com.youth.banner:banner:1.4.10'
//是否需要扫码：
    implementation 'com.dlazaro66.qrcodereaderview:qrcodereaderview:2.0.3'
//是否需要内部浏览器，需要的话X5内核：
    implementation files('libs/tbs_sdk_thirdapp_v3.2.0.1104_43200_sharewithdownload_withfilereader_withoutGame_obfs_20170609_115346.jar')
}

<uses-permission android:name="android.permission.ACCESS_FINE_LOCATION" />

<!-- 更新时需要的安装权限 -->
<uses-permission android:name="android.permission.REQUEST_INSTALL_PACKAGES"/>

额外的，临时目录申请权限：

<provider
    android:name="android.support.v4.content.FileProvider"
    android:authorities="应用包名.fileProvider"
    android:exported="false"
    android:grantUriPermissions="true">
    <meta-data
        android:name="android.support.FILE_PROVIDER_PATHS"
        android:resource="@xml/file_path" />
</provider>

file_path.xml：

<resources xmlns:android="http://schemas.android.com/apk/res/android">
    <paths>
        <external-path path="." name="external_storage_root" />
        <external-path path="" name="download"/>
    </paths>
</resources>

基础网络请求：

HttpRequest.getInstance(me)
        //自定义请求Header头部信息（选用）
        .setHeaders(new Parameter()
                .add("Charset", "UTF-8")
                .add("Content-Type", "application/json")
                .add("Accept-Encoding", "gzip,deflate")
        )
        //发送请求
        .postRequest("http://www.xxx.com/test", new Parameter()
                        .add("key1", "value1")
                        .add("key2", "value3")
                        .add("key4", "value4"),
                new ResponseListener() {
                    @Override
                    public void onResponse(JSONObject main, Exception error) {
                        //处理返回数据逻辑
                        if (error == null) {
                            //请求成功处理
                        } else {
                            //请求失败处理
                            Toast.makeText(me, "网络错误，请重试", Toast.LENGTH_SHORT);
                        }
                    }
                });
